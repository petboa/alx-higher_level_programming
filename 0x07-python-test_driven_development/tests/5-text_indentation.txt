import text_indentation

assert text_indentation.text_indentation("poop") == "poop"

assert text_indentation.text_indentation("po.op") == "po.\n\nop"

assert text_indentation.text_indentation("po: op") == "po:\n\nop"

assert text_indentation.text_indentation("po.     op") == "po.\n\nop"

assert text_indentation.text_indentation("") is None

assert text_indentation.text_indentation("poop:      ?    ") == "poop:\n\n?"

assert text_indentation.text_indentation("hello:      ?    ") == "hello:\n\n?"

assert text_indentation.text_indentation("hi...") == "hi.\n\n.\n\n."

assert text_indentation.text_indentation("             test     ") == "test"

# The following lines test for errors
try:
    text_indentation.text_indentation(3)
except TypeError:
    pass
else:
    raise AssertionError("Expected TypeError")

try:
    text_indentation.text_indentation(None)
except TypeError:
    pass
else:
    raise AssertionError("Expected TypeError")
